"use strict"

/**
 * @module gh
 */

/**
 * @class gh
 */
var gh = (function(gh){
	console.log("door.js loaded");

	/**
	 * @class Door
	 * @constructor
	 * @param {string} side Values can include "top" "left" "right" "bottom".
	 * @param {bool} open
	 */
	function Door(side, open, imageStates){
		this.side 			= side;
		this.open 			= open || false;

		this.focus 			= false; // Is mouse currently over the door.
		this.imageStates	= imageStates;
	}

	/**
	 * @method draw
	 * @param {Canvas.context} context
	 * @param {integer} x
	 * @param {integer} y
	 * @param {float} tileSize
	 * @param {float} scale
	 */
	Door.prototype.draw = function(context, x, y, tileSize, scale){
		var img;

		// Determine which image for display based on the door's state.
		if(this.open){
			if(this.focus){
				img = this.imageStates.openDoorHighlight;
			} else {
				img = this.imageStates.open;
			}
		} else {
			if(this.focus){
				img = this.imageStates.closedHighlight;
			} else {
				img = this.imageStates.closed;
			}
		}

		context.save();

		switch(this.side){
			case "top":
				gh.assets.sprites[img].draw(
					context,
					x * tileSize * scale,
					y * tileSize * scale - (0.5 * scale * gh.assets.sprites[img].img.height),
					tileSize * scale,
					gh.assets.sprites[img].img.height * scale
				);
				break;
			case "left":
				context.translate(x * tileSize * scale, y * tileSize * scale);
				context.rotate(math.degToRadians(90));
				gh.assets.sprites[img].draw(
					context,
					0,
					0 - (0.5 * scale * gh.assets.sprites[img].img.height),
					tileSize * scale,
					gh.assets.sprites[img].img.height * scale
				);
				break;
			default:
				break;
		}

		context.restore();

		return;
	};

	gh.Door = Door;

	return gh;
})(gh || {});
