"use strict"

/**
 * @module gh
 */

/**
 * @class gh
 */
var gh = (function(gh){

	console.log("agent.js loaded");

	var state = {};
	state.INERT 		= 0;
	state.MOVE 			= 1;
	state.DEAD 			= 2;
	state.FOCUS 		= 3;
	state.TURN 			= 4;

	/**
	 * @class Agent
	 * @constructor
	 * @param {string} templateID
	 * @param {string} uniqueID
	 * @param {string} description
	 * @param {integer} x
	 * @param {integer} y
	 * @param {} team
	 * @param {integer} body
	 * @param {integer} mind
	 * @param {} baseDefence
	 * @param {} mainHand
	 * @param {} offHand
	 * @param {} chest
	 * @param {} head
	 * @param {} moveDice
	 * @param {} baseMove
	 * @param {} spellList
	 * @param {} inventory
	 */
	function Agent(templateID,uniqueID, x, y, team, description, body, mind, baseDefence, mainHand, offHand, chest, head, moveDice, baseMove, spellList, inventory){
		this.templateID 		= templateID;
		this.uniqueID 			= uniqueID;
		this.x 					= x || -1;
		this.y 					= y || -1;
		this.team				= team || "";
		this.description		= description || "";
		this.health 			= {"max" : body || 0, "damage" : 0};
		this.mind 				= {"max" : mind || 0, "damage" : 0};
		this.baseDefence 		= baseDefence || 0;
		this.mainHand 		 	= mainHand || null;
		this.offHand 			= offHand || null;
		this.chest				= chest || null;
		this.head				= head || null;
		this.moveDice			= moveDice || 0;
		this.baseMove			= baseMove || 0;
		this.spellList			= spellList || null;
		this.inventory			= inventory || null;

		this.actionState		= "inert";

		return true;
	}

	gh.Agent = Agent;

	return gh;
})(gh || {});
